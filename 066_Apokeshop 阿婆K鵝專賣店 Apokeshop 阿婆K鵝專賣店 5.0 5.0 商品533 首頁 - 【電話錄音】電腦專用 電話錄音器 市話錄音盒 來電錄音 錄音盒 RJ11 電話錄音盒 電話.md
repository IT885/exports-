        if self.recording:

## 🎵 方案二：直接監聽RTP流（進階）

### **UDP RTP流監聽錄音**[2]
```python
import socket
import struct
import wave
import threading
from datetime import datetime

class RTPRecorder:
    def __init__(self, listen_ip="0.0.0.0", listen_port=5004):
        self.listen_ip = listen_ip
        self.listen_port = listen_port
        self.recording = False
        self.audio_buffer = []
        self.socket = None
        
    def start_listening(self):
        """開始監聽RTP流"""
        self.socket = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
        self.socket.bind((self.listen_ip, self.listen_port))
        self.recording = True
        
        print(f"開始監聽 {self.listen_ip}:{self.listen_port}")
        
        while self.recording:
            try:
                data, addr = self.socket.recvfrom(1024)
                rtp_packet = self.parse_rtp(data)
                
                if rtp_packet and rtp_packet['payload']:
                    self.audio_buffer.append(rtp_packet['payload'])
                    print(f"收到音頻包，來源: {addr}")
                    
            except socket.timeout:
                continue
            except Exception as e:
                print(f"錯誤: {e}")
                
    def parse_rtp(self, data):
        """解析RTP包"""
        if len(data) < 12:  # RTP最小包長度
            return None
            
        # RTP包頭解析
        version = (data[0] & 0xC0) >> 6
        padding = (data[0] & 0x20) >> 5
        extension = (data[0] & 0x10) >> 4
        cc = data[0] & 0x0F
        
        marker = (data[1] & 0x80) >> 7
        payload_type = data[1] & 0x7F
        
        sequence_number = struct.unpack('!H', data[2:4])[0]
        timestamp = struct.unpack('!I', data[4:8])[0]
        ssrc = struct.unpack('!I', data[8:12])[0]
        
        # 音頻載荷開始位置
        header_length = 12 + (cc * 4)
        payload = data[header_length:]
        
        return {
            'version': version,
            'payload_type': payload_type,
            'sequence_number': sequence_number,
            'timestamp': timestamp,
            'ssrc': ssrc,
            'payload': payload
        }
        
    def save_recording(self, filename):
        """保存錄音檔案"""
        if not self.audio_buffer:
            print("沒有音頻數據可保存")
            return
            
        with wave.open(filename, 'wb') as wf:
            wf.setnchannels(1)  # 單聲道
            wf.setsampwidth(2)  # 16位元
            wf.setframerate(8000)  # G.711標準取樣率
            wf.writeframes(b''.join(self.audio_buffer))
            
        print(f"錄音已保存: {filename}")
        
    def stop_recording(self):
        """停止錄音"""
        self.recording = False
        if self.socket:
            self.socket.close()
```
