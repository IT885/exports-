### 1. 郵件內容清理模組

```python
import re
import email
from bs4 import BeautifulSoup
from email.mime.multipart import MIMEMultipart
import html2text

class EmailPreprocessor:
    def __init__(self):
        # 常見簽名檔模式
        self.signature_patterns = [
            r'Best regards,.*?(?=\n\n|\n-{3,}|\Z)',
            r'此致\s*敬禮.*?(?=\n\n|\n-{3,}|\Z)',
            r'TEL：.*?(?=\n\n|\n-{3,}|\Z)',
            r'E-mail：.*?(?=\n\n|\n-{3,}|\Z)',
            r'地址：.*?(?=\n\n|\n-{3,}|\Z)',
            r'客服專線：.*?(?=\n\n|\n-{3,}|\Z)',
            r'═{3,}.*?(?=\n\n|\Z)',  # 您的郵件中的分隔線
        ]
        
        # 回覆鏈分割模式
        self.reply_chain_patterns = [
            r'From:.*?Sent:.*?To:.*?Subject:',
            r'寄件者:.*?寄件日期:.*?收件者:.*?主旨:',
            r'-----Original Message-----',
            r'---- 原始郵件 ----',
        ]
    
    def clean_html(self, html_content):
        """清理HTML標籤，保留純文字"""
        soup = BeautifulSoup(html_content, 'html.parser')
        
        # 移除script和style標籤
        for script in soup(["script", "style"]):
            script.decompose()
        
        # 轉換為純文字
        h = html2text.HTML2Text()
        h.ignore_links = True
        h.ignore_images = True
        return h.handle(str(soup))
    
    def remove_signatures(self, content):
        """移除郵件簽名檔"""
        cleaned = content
        for pattern in self.signature_patterns:
            cleaned = re.sub(pattern, '', cleaned, flags=re.DOTALL | re.IGNORECASE)
        return cleaned.strip()
    
    def split_reply_chain(self, content):
        """分割回覆鏈，提取各層級郵件"""
        emails = []
        current_content = content
        
        for pattern in self.reply_chain_patterns:
            match = re.search(pattern, current_content, re.IGNORECASE | re.DOTALL)
            if match:
                # 提取當前郵件內容
                current_email = current_content[:match.start()].strip()
                if current_email:
                    emails.append({
                        'content': current_email,
                        'is_current': len(emails) == 0
                    })
                
                # 繼續處理剩餘部分
                current_content = current_content[match.start():]
                
        # 如果沒有找到分割點，整封都是當前郵件
        if not emails and current_content.strip():
            emails.append({
                'content': current_content.strip(),
                'is_current': True
            })
            
        return emails
```

### 3. 完整處理流程

```python
class EmailProcessingPipeline:
    def __init__(self, gemini_api_key: str):
        self.preprocessor = EmailPreprocessor()
        self.analyzer = GeminiEmailAnalyzer(gemini_api_key)
        
    def process_single_email(self, raw_email: str) -> Dict:
        """處理單封郵件的完整流程"""
        result = {
            "original_length": len(raw_email),
            "processing_steps": [],
            "analysis": None,
            "errors": []
        }
        
        try:
            # Step 1: HTML清理
            cleaned_html = self.preprocessor.clean_html(raw_email)
            result["processing_steps"].append("HTML清理完成")
            
            # Step 2: 簽名檔移除
            no_signature = self.preprocessor.remove_signatures(cleaned_html)
            result["processing_steps"].append("簽名檔移除完成")
            
            # Step 3: 回覆鏈分割
            email_chain = self.preprocessor.split_reply_chain(no_signature)
            result["email_chain_count"] = len(email_chain)
            result["processing_steps"].append(f"回覆鏈分割完成，共{len(email_chain)}層")
            
            # Step 4: 分析最新郵件（通常是第一層）
            if email_chain:
                current_email = email_chain[0]['content']
                result["cleaned_content"] = current_email
                result["content_reduction"] = f"{len(raw_email)} → {len(current_email)} 字元"
                
                # Step 5: Gemini API 分析
                analysis = self.analyzer.analyze_email(current_email)
                result["analysis"] = analysis
                result["processing_steps"].append("Gemini分析完成")
            
        except Exception as e:
            result["errors"].append(str(e))
            
        return result
    
    def batch_process_emails(self, email_list: List[str]) -> List[Dict]:
        """批次處理郵件列表"""
        results = []
        
        for i, email_content in enumerate(email_list):
            print(f"處理第 {i+1}/{len(email_list)} 封郵件...")
            
            result = self.process_single_email(email_content)
            result["email_index"] = i
            results.append(result)
            
            # 避免超過API限制
            if i % 50 == 0:  # 每50封郵件暫停
                print("暫停60秒避免超過API限制...")
                time.sleep(60)
                
        return results
```
