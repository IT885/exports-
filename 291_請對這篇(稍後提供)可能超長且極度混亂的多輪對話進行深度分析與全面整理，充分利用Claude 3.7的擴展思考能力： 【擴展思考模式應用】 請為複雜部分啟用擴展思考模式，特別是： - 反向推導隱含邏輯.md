const ChatPage = {
  template: `
    <div class="chat-container">
      <!-- 使用 ant-design-x-vue 的聊天組件 -->
      <ax-bubble-list
        :roles="roles"
        :items="messages.map(({id, message, status}) => ({

          loading: status === 'loading',
          role: status === 'local' ? 'local' : 'ai',

        :value="content"
        @change="setContent"

      ></ax-sender>

  `,
  setup() {
    const content = ref('');
    const senderLoading = ref(false);
    
    // 使用 ant-design-x-vue 的 useXAgent
    const [agent] = AntDesignX.useXAgent({
      request: async ({ message }, { onSuccess, onUpdate }) => {
        senderLoading.value = true;
        
        // 模擬流式輸出
        const fullContent = `您好！這是對您問題「${message}」的回覆...`;
        let currentContent = '';
        
        const id = setInterval(() => {
          currentContent = fullContent.slice(0, currentContent.length + 3);
          onUpdate(currentContent);
          
          if (currentContent === fullContent) {
            clearInterval(id);
            senderLoading.value = false;
            onSuccess(fullContent);
          }
        }, 100);
      }
    });
    
    // 角色配置
    const roles = {
      ai: {
        placement: 'start',
        avatar: { icon: '<el-icon><User /></el-icon>', style: { background: '#fde3cf' } },
      },
      local: {
        placement: 'end',
        avatar: { icon: '<el-icon><User /></el-icon>', style: { background: '#87d068' } },
      }
    };
    
    // 使用 useXChat 管理對話
    const { onRequest, messages } = AntDesignX.useXChat({
      agent: agent.value,
      requestPlaceholder: '思考中...',
    });
    
    const handleSubmit = (nextContent) => {
      onRequest(nextContent);
      content.value = '';
    };
    
    const setContent = (v) => {
      content.value = v;
    };
    
    return {
      content,
      senderLoading,
      messages,
      roles,
      handleSubmit,
      setContent
    };
  }
