```python
def table_to_markdown(table):
    rows = table.find_all('tr')
    markdown = []
    for i, row in enumerate(rows):
        cols = row.find_all(['td', 'th'])
        cols_text = [col.get_text(strip=True) for col in cols]
        markdown.append('| ' + ' | '.join(cols_text) + ' |')
        if i == 0:  # 標題行
            markdown.append('| ' + ' | '.join(['---'] * len(cols_text)) + ' |')
    return '\n'.join(markdown)
```

```python
def extract_paragraphs(soup):
```

```python
def parse_email_html(html_content):
    """解析郵件HTML內容"""
    # 使用lxml作為後端解析器（速度較快）
    soup = BeautifulSoup(html_content, 'lxml')
    
    # 移除不需要的標籤
    for tag in soup(['script', 'style', 'meta', 'link', 'head']):
        tag.decompose()
    
    return soup
```

#### 步驟2：表格處理
```python
def extract_tables(soup):
    """提取並轉換表格為Markdown格式"""
    tables_data = []
    
    for table in soup.find_all('table'):
        rows = table.find_all('tr')
        table_content = []
        
        for row in rows:
            cells = row.find_all(['td', 'th'])
            row_data = [cell.get_text(strip=True) for cell in cells]
            table_content.append(row_data)
        
        # 轉換為Markdown表格格式
        if table_content:
            markdown_table = convert_to_markdown_table(table_content)
            tables_data.append(markdown_table)
    
    return tables_data

def convert_to_markdown_table(table_data):
    """轉換為Markdown表格格式"""
    if not table_data:
        return ""
    
    markdown = []
    for i, row in enumerate(table_data):
        markdown.append('| ' + ' | '.join(row) + ' |')
        if i == 0:  # 標題行後加分隔線
            markdown.append('| ' + ' | '.join(['---'] * len(row)) + ' |')
    
    return '\n'.join(markdown)
```

#### 步驟3：段落與聯絡資訊提取
```python
def extract_paragraphs(soup):
    """提取段落內容"""
    paragraphs = []
    
    # 提取段落
    for p in soup.find_all('p'):
        text = p.get_text(strip=True)
        if text and len(text) > 1:  # 過濾空段落
            paragraphs.append(text)
    
    return paragraphs

def extract_contact_info(text):
    """提取聯絡資訊"""
    contact_patterns = {
        'email': r'[\w\.-]+@[\w\.-]+\.\w+',
        'phone': r'[\d\-\+\(\)\s]{8,}',
        'address': r'[\u4e00-\u9fff]+(市|縣|區|路|街|號)[\w\-\s]*'
    }
    
    contact_info = {}
    for key, pattern in contact_patterns.items():
        matches = re.findall(pattern, text)
        if matches:
            contact_info[key] = matches
    
    return contact_info
```

```python
def process_email_html(email_json):
    """完整的郵件HTML處理流程"""
    
    # 取得HTML內容
    html_content = email_json.get('body', {}).get('content', '')
    
    if not html_content:
        return None
    
    # 解析HTML
    soup = parse_email_html(html_content)
    
    # 提取各種內容
    tables = extract_tables(soup)
    paragraphs = extract_paragraphs(soup)
    
    # 取得純文字內容
    clean_text = soup.get_text(separator='\n', strip=True)
    
    # 提取聯絡資訊
    contact_info = extract_contact_info(clean_text)
    
    # 組織結果
    processed_data = {
        'email_metadata': {
            'sender': email_json.get('from', ''),
            'recipients': email_json.get('to', []),
            'cc': email_json.get('cc', []),
            'subject': email_json.get('subject', ''),
            'date': email_json.get('sentDateTime', '')
        },
        'content': {
            'clean_text': clean_text,
            'paragraphs': paragraphs,
            'tables': tables,
            'contact_info': contact_info
        },
        'structure_analysis': {
            'has_table': len(tables) > 0,
            'paragraph_count': len(paragraphs),
            'has_contact_info': len(contact_info) > 0
        }
    }
    
    return processed_data
```

```python
# html_processor.py
from bs4 import BeautifulSoup
import re

class EmailHTMLProcessor:
    def __init__(self):
        self.processing_stats = {
            "processed_count": 0,
            "tables_extracted": 0,
            "contact_info_found": 0
        }
    
    def process_html_content(self, html_content):
        """處理HTML內容 - 不保留原始HTML"""
        if not html_content:
            return None
            
        # 使用lxml作為後端解析器
        soup = BeautifulSoup(html_content, 'lxml')
        
        # 移除不需要的標籤
        for tag in soup(['script', 'style', 'meta', 'link', 'head']):
            tag.decompose()
        
        # 提取結構化內容
        return {
            "clean_text": soup.get_text(separator='\n', strip=True),
            "paragraphs": self.extract_paragraphs(soup),
            "tables": self.extract_tables_as_markdown(soup),
            "contact_info": self.extract_contact_info(soup),
            "has_important_content": self.validate_content_quality(soup)
        }
    
    def extract_paragraphs(self, soup):
        """提取段落內容"""
        paragraphs = []
        for p in soup.find_all('p'):
            text = p.get_text(strip=True)
            if text and len(text) > 1:
                paragraphs.append(text)
        return paragraphs
    
    def extract_tables_as_markdown(self, soup):
        """提取表格並轉換為Markdown格式"""
        tables_data = []
        for table in soup.find_all('table'):
            rows = table.find_all('tr')
            table_content = []
            
            for row in rows:
                cells = row.find_all(['td', 'th'])
                row_data = [cell.get_text(strip=True) for cell in cells]
                if any(row_data):  # 只保留非空行
                    table_content.append(row_data)
            
            if table_content:
                markdown_table = self.convert_to_markdown_table(table_content)
                tables_data.append(markdown_table)
                self.processing_stats["tables_extracted"] += 1
                
        return tables_data
    
    def convert_to_markdown_table(self, table_data):
        """轉換為Markdown表格格式"""
        if not table_data:
            return ""
        
        markdown = []
        for i, row in enumerate(table_data):
            markdown.append('| ' + ' | '.join(row) + ' |')
            if i == 0:  # 標題行後加分隔線
                markdown.append('| ' + ' | '.join(['---'] * len(row)) + ' |')
        
        return '\n'.join(markdown)
    
    def extract_contact_info(self, soup):
        """提取聯絡資訊"""
        text_content = soup.get_text()
        
        contact_patterns = {
            'email': r'[\w\.-]+@[\w\.-]+\.\w+',
            'phone': r'[\d\-\+\(\)\s]{8,}',
            'address': r'[\u4e00-\u9fff]+(市|縣|區|路|街|號)[\w\-\s]*'
        }
        
        contact_info = {}
        for key, pattern in contact_patterns.items():
            matches = re.findall(pattern, text_content)
            if matches:
                contact_info[key] = matches
                
        if contact_info:
            self.processing_stats["contact_info_found"] += 1
            
        return contact_info
    
    def validate_content_quality(self, soup):
        """驗證內容品質"""
        text_content = soup.get_text(strip=True)
        return len(text_content) > 10  # 基本內容長度檢查
```

```python
        processed_content = self.html_processor.process_html_content(html_content)
        
        if not processed_content:
            return None
        
        # AI智能分類
        classification = self.ai_classifier.classify_email_with_ai(processed_content['clean_text'])
        
        # 組織最終結果
        email_id = self.generate_email_id()
        final_result = {
            "email_id": email_id,
            "processing_info": {
                "processed_date": datetime.now().isoformat(),
                "source_file": json_file_path,
                "processing_stage": "stage_1"
            },
            "original_metadata": {
                "sender": email_json.get('from', {}).get('emailAddress', {}).get('address', ''),
                "subject": email_json.get('subject', ''),
                "received_date": email_json.get('receivedDateTime', ''),
                "recipients": [r.get('emailAddress', {}).get('address', '') for r in email_json.get('toRecipients', [])]
            },
            "content": processed_content,
            "classification": classification,
            "quality_check": {
                "content_completeness": 100 if processed_content['has_important_content'] else 50,
                "requires_manual_review": classification.get('confidence', 0) < 70
            }
        }
        
        # 儲存處理結果
        self.file_manager.save_processed_email(final_result, email_id)
        
        return final_result

            "classification_stats": {},

            }
```

### 類別（Category）與旗標（Flag）在 AI 智能標記的實際應用說明

#### 1. 類別（Category）－ 白話舉例與應用

- **什麼是類別？**  
  就像在辦公桌上貼顏色標籤一樣，Outlook 的「類別」可以自訂名稱與顏色，幫每封郵件分門別類。
- **AI 標記應用舉例：**  
  - 假設你有一個資料夾「專案A」，AI 可自動將所有與「專案A」有關的郵件加上「專案A」的藍色類別。
  - 收到來自特定客戶的郵件時，AI 自動加上「VIP 客戶」的紅色類別，方便優先處理。
  - AI 依據郵件內容自動判斷主題，標記「需審核」、「待回覆」、「會議通知」等類別，日後搜尋或篩選時一目了然。
- **資料夾位置與關聯：**  
  - 類別標記與郵件所在資料夾無關，可跨資料夾搜尋。例如同一類別的郵件可能分散在「收件匣」、「已寄信件」、「專案資料夾」等不同位置，但都能用類別快速找出。

#### 2. 旗標（Flag）－ 白話舉例與應用

- **什麼是旗標？**  
  旗標就像給郵件插上「提醒」小旗子，通常用來標記待辦事項或需要後續處理的郵件。
- **AI 標記應用舉例：**  
  - AI 根據郵件內容自動判斷哪些郵件需要跟進，幫你加上旗標，並設定提醒日期（例如三天內需回覆）。
  - 收到需要簽核的郵件時，AI 自動加上旗標並註記「需簽核」，這樣郵件會自動出現在 Outlook 的「待辦事項」清單中。
  - AI 可根據郵件重要性自動設定不同旗標註記（如「高優先」、「一般」），協助你分辨處理順序。
- **資料夾位置與關聯：**  
  - 旗標標記同樣不受資料夾限制。無論郵件被移到哪個資料夾，只要加上旗標，就能在「待辦事項」或「旗標郵件」視圖中統一管理。
  - 旗標也可搭配 AI 記錄相關說明（如「此郵件需與XX部門確認」），方便後續追蹤。

#### 3. 類別與旗標在 AI 管理郵件時的優勢

| 功能      | 說明                                   | 舉例應用                                       |
|-----------|----------------------------------------|------------------------------------------------|
| 類別      | 分類、標記主題或專案，支援多重分類     | AI 標記「專案A」、「需回覆」等，方便搜尋         |
| 旗標      | 標記待辦、提醒、跟進，支援註記         | AI 加旗標「需簽核」，自動進入待辦清單           |
| 跨資料夾  | 不受資料夾限制，搜尋與篩選皆適用       | 不論郵件在哪個資料夾，加上類別/旗標都能找得到   |
| 關聯管理  | 可搭配 AI 記錄關聯資訊於旗標註記或內容 | 旗標註記「與XX部門協作」，或內容加上關聯說明    |

#### 4. 實務建議
