```python
def expand_query(query):
    """擴展查詢關鍵字，返回擴展後的查詢列表"""
    expanded_queries = [query]  # 原始查詢
    
    # 基本分詞
    terms = query.split()
    if len(terms) > 1:
        expanded_queries.extend(terms)
    
    # 領域特定擴展
    domain_expansions = {
        "美食": ["餐廳", "食物", "料理", "小吃"],

        # 可添加更多領域
    }
    
    # 根據領域生成新查詢
    for term in terms:
        if term in domain_expansions:
            for expansion in domain_expansions[term]:
                expanded_queries.append(query.replace(term, expansion))
    
    # 特定組合的語意擴展
    if "金牛座" in query and "美食" in query:
        expanded_queries.extend([
            "金牛座喜歡的食物", "適合金牛座的餐廳", "金牛座飲食偏好"
```

```python
# MCP服務器功能
class MCPWebCrawler:
    """MCP服務器爬蟲工具"""
    
    def __init__(self, name="web_crawler"):
        self.name = name
    
    def search(self, query, engines=None, max_results=5, expand_queries=False):
        """搜索功能（MCP工具）"""
        if not engines:
            engines = ["google"]
        
        results = []
        for engine in engines:
            result = search_with_engine(query, engine, max_results, 
                                        expand_queries=expand_queries)
            results.append(result)
        
        return {
            "query": query,
            "timestamp": datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S'),
            "engines": engines,
            "results": results
        }
    
    def get_related_keywords(self, query):
        """獲取相關關鍵字（MCP工具）"""
        result = search_with_engine(query, "google", max_results=1)
        related_searches = result.get("related_searches", [])
        expanded_queries = expand_query(query)
        
        return {
            "query": query,
            "related_keywords": list(set(related_searches + expanded_queries))
        }
```

```python
                                    "snippet": snippet,
                                    "query": current_query
                                })
                
                print(f"{search_engine} 使用查詢 '{current_query}' 找到 {len(results)} 個搜索結果")
                all_results.extend(results)
                
                # 限制每個查詢的結果，如果已經達到最大結果數則停止查詢
                if len(all_results) >= max_results:
                    break
            
            # 限制總結果數量
            all_results = all_results[:max_results]
            
            # 訪問結果頁面獲取更多內容
            detailed_results = []
            for i, result in enumerate(all_results):
                try:
                    print(f"{search_engine} 正在訪問結果 {i+1}/{len(all_results)}: {result['title']}")
                    page.goto(result["link"], timeout=30000)
                    
                    # 等待頁面加載
                    try:
                        page.wait_for_load_state("networkidle", timeout=10000)
                    except:
                        print(f"{search_engine} 頁面加載超時，繼續處理...")
                    
                    # 模擬人類行為：隨機滾動
                    for _ in range(random.randint(2, 4)):
                        page.mouse.wheel(0, random.randint(300, 700))
                        time.sleep(0.3 + random.random() * 0.5)
                    
                    # 截圖（調試模式）
                    if debug:
                        page.screenshot(path=f"debug_screenshots/{search_engine}_result_{i+1}_{timestamp}.png")
                    
                    page_content = page.content()
                    
                    # 保存完整內容
                    if save_full_content:
                        site_name = re.sub(r'[\\/*?:"<>|]', "_", result["title"])
                        with open(f"full_content/{search_engine}_{site_name}_{timestamp}.html", "w", encoding="utf-8") as f:
                            f.write(page_content)
                    
                    page_soup = BeautifulSoup(page_content, 'html.parser')
                    
                    # 提取頁面文本
                    paragraphs = []
                    
                    # 先檢查文章主體
                    article = page_soup.find('article')
                    if article:
                        elements = article.find_all(['p', 'h1', 'h2', 'h3', 'h4', 'li'])
                    else:
                        # 嘗試找到主要內容區域
                        main_content = page_soup.find('main') or page_soup.find(id='content') or page_soup.find(class_='content')
                        if main_content:
                            elements = main_content.find_all(['p', 'h1', 'h2', 'h3', 'h4', 'li'])
                        else:
                            elements = page_soup.find_all(['p', 'h1', 'h2', 'h3', 'h4'])
                    
                    # 提取段落
                    for element in elements:
                        # 跳過導航、頁腳等元素
                        if element.parent and element.parent.name in ['nav', 'footer', 'header']:
                            continue
                        
                        # 跳過可能的菜單元素
                        if element.get('class') and any(c in str(element.get('class')) for c in ['menu', 'nav', 'footer', 'header']):
                            continue
                        
                        text = element.get_text().strip()
                        if len(text) > 50:  # 忽略太短的文本
                            # 檢查是否包含查詢關鍵詞
                            keywords = query.lower().split()
                            if any(kw in text.lower() for kw in keywords):
                                paragraphs.append(text)
                    
                    # 如果沒找到相關段落，取前幾段
                    if not paragraphs:
                        for p in page_soup.find_all('p'):
                            text = p.get_text().strip()
                            if len(text) > 100:
                                paragraphs.append(text)
                            if len(paragraphs) >= 3:
                                break
                    
                    # 提取發布日期
                    publish_date = None
                    date_meta = page_soup.find('meta', {'property': 'article:published_time'}) or \
                                page_soup.find('meta', {'name': 'pubdate'}) or \
                                page_soup.find('meta', {'name': 'date'})
                    if date_meta and date_meta.get('content'):
                        publish_date = date_meta['content']
                    
                    # 提取作者
                    author = None
                    author_meta = page_soup.find('meta', {'name': 'author'}) or \
                                 page_soup.find('meta', {'property': 'article:author'})
                    if author_meta and author_meta.get('content'):
                        author = author_meta['content']
                    
                    result["paragraphs"] = paragraphs[:5]  # 最多保存5段相關文本
                    result["publish_date"] = publish_date
                    result["author"] = author
                    
                    detailed_results.append(result)
                
                except Exception as e:
                    print(f"{search_engine} 訪問結果時出錯: {e}")
                    # 仍然添加基本信息
                    result["paragraphs"] = []
                    detailed_results.append(result)
            
            browser.close()
            
            return {
                "engine": search_engine,
                "query": query,
                "timestamp": datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S'),
                "results_count": len(detailed_results),
                "results": detailed_results,
                "related_searches": list(set(related_searches_all)),
                "expanded_queries": queries_to_search if expand_queries else []
            }
    
    except Exception as e:
        print(f"{search_engine} 搜索過程中發生錯誤: {e}")
        return {
            "engine": search_engine,
            "query": query,
            "timestamp": datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S'),
            "results_count": 0,
            "error": str(e),
            "results": [],
            "related_searches": [],
            "expanded_queries": queries_to_search if expand_queries else []
        }

def search_and_save_as_md(query, engines=None, max_results=5, debug=False, save_full_content=False, expand_queries=False):
    """使用多個搜索引擎搜索並將結果保存為Markdown"""
    print(f"開始搜索: {query}")
    
    # 創建results目錄（如果不存在）
    if not os.path.exists("markdown_results"):
        os.makedirs("markdown_results")
    
    timestamp = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
    filename = f"markdown_results/search_results_{clean_filename(query)}_{timestamp}.md"
    
    # 如果未指定搜索引擎，使用默認值
    if not engines:
        engines = ["google", "bing"]
    
    # 使用線程池並行搜索
    all_results = []
    with concurrent.futures.ThreadPoolExecutor(max_workers=len(engines)) as executor:
        # 創建任務
        future_to_engine = {
            executor.submit(search_with_engine, query, engine, max_results, debug, save_full_content, expand_queries): engine
            for engine in engines
        }
        
        # 獲取結果
        for future in concurrent.futures.as_completed(future_to_engine):
            engine = future_to_engine[future]
            try:
                result = future.result()
                all_results.append(result)
                print(f"{engine} 搜索完成，找到 {result['results_count']} 個結果")
            except Exception as e:
                print(f"{engine} 搜索失敗: {e}")
    
    # 收集所有相關搜索
    all_related_searches = []
    for result in all_results:
        if "related_searches" in result:
            all_related_searches.extend(result["related_searches"])
    
    # 保存為Markdown
    with open(filename, "w", encoding="utf-8") as f:
        f.write(f"# 多引擎搜索結果: {query}\n")
        f.write("=========================\n\n\n")
        
        f.write(f"搜索查詢: **{query}**\n\n")
        f.write(f"搜索時間: {datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S')}\n\n")
        f.write(f"搜索引擎: **{', '.join(engines)}**\n\n")
        
        # 統計總結果數
        total_results = sum(r["results_count"] for r in all_results)
        f.write(f"總結果數量: **{total_results}**\n\n")
        
        # 如果啟用了查詢擴展，顯示擴展的查詢
        if expand_queries:
            expanded_queries = []
            for result in all_results:
                if "expanded_queries" in result:
                    expanded_queries.extend(result["expanded_queries"])
            if expanded_queries:
                f.write("## 擴展查詢\n\n")
                for eq in set(expanded_queries):
                    f.write(f"- {eq}\n")
                f.write("\n")
        
        # 顯示相關搜索
        if all_related_searches:
            f.write("## 相關搜索\n\n")
            for rs in set(all_related_searches):
                f.write(f"- {rs}\n")
            f.write("\n")
        
        if total_results == 0:
            f.write("**未找到相關結果**\n\n")
        
        # 為每個搜索引擎寫入結果
        for engine_result in all_results:
            engine = engine_result["engine"]
            results = engine_result["results"]
            
            f.write(f"## {engine.upper()} 搜索結果 ({len(results)})\n\n")
            
            if "error" in engine_result and engine_result["error"]:
                f.write(f"**錯誤**: {engine_result['error']}\n\n")
            
            for i, result in enumerate(results):
                f.write(f"### {i+1}. {result['title']}\n\n")
                f.write(f"**鏈接**: [{result['link']}]({result['link']})\n\n")
                
                if "query" in result and result["query"] != query:
                    f.write(f"**使用查詢**: {result['query']}\n\n")
                
                if result.get("author"):
                    f.write(f"**作者**: {result['author']}\n\n")
                
                if result.get("publish_date"):
                    f.write(f"**發布日期**: {result['publish_date']}\n\n")
                
                f.write(f"**摘要**: {result['snippet']}\n\n")
                
                if result.get("paragraphs") and len(result["paragraphs"]) > 0:
                    f.write("**相關內容**:\n\n")
                    for p in result["paragraphs"]:
                        f.write(f"> {p}\n\n")
                
                f.write("---\n\n")
            
            f.write("\n")
        
        f.write("\n*搜索結果由多引擎網絡爬蟲自動生成*")
    
    # 同時保存為JSON格式（方便程序讀取）
    json_filename = f"markdown_results/search_results_{clean_filename(query)}_{timestamp}.json"
    with open(json_filename, "w", encoding="utf-8") as f:
        json.dump({
            "query": query,
            "timestamp": datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S'),
            "engines": engines,
            "results": all_results,
            "related_searches": list(set(all_related_searches))
        }, f, ensure_ascii=False, indent=2)
    
    print(f"搜索結果已保存到: {filename} 和 {json_filename}")
    return filename

def start_mcp_server(host="localhost", port=7777):
    """啟動MCP服務器"""
    try:
        from mcp import MCPServer
    except ImportError:
        print("未安裝MCP庫。請使用 'pip install mcp-api-sdk' 安裝。")
        return
    
    crawler = MCPWebCrawler()
    server = MCPServer(
        tools=[crawler],
        host=host,
        port=port
    )
    print(f"啟動MCP服務器在 {host}:{port}...")
    server.start()
```
