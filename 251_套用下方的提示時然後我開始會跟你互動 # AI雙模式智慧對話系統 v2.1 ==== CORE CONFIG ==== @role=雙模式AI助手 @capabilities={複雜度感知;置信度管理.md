```python
class ContentDeduplication:
    def __init__(self):

        self.processed_fingerprints = set()
        
    def check_duplicate(self, new_post):
        # 多層次去重檢查
        
        # 1. 完全相同內容
        content_hash = hashlib.md5(new_post.content.encode()).hexdigest()
        if content_hash in self.processed_fingerprints:
            return True, "exact_duplicate"

        similarity_score = self.calculate_similarity(new_post)
        if similarity_score > self.similarity_threshold:
            return True, "similar_content"
            
        # 3. 重新分享檢測
        if self.is_reshare(new_post):
            return True, "reshare_duplicate"
            
        return False, "unique_content"
    
    def calculate_similarity(self, post):
        # TF-IDF + 餘弦相似度
        return self.similarity_engine.compare(post.content)
```

```python
class VectorKnowledgeRetrieval:
    def __init__(self):
        self.embedding_model = SentenceTransformers('all-MiniLM-L6-v2')
        self.vector_db = ChromaDB()
    
    def build_personal_vector_db(self, user_knowledge):
        # 建立個人化向量資料庫
        for knowledge_item in user_knowledge:
            embedding = self.embedding_model.encode(knowledge_item.content)
            self.vector_db.add(
                embeddings=[embedding],
                documents=[knowledge_item.content],
                metadatas=[knowledge_item.metadata],
                ids=[knowledge_item.id]
            )
    
    def semantic_search(self, query, user_id):
        # 語義搜尋個人知識庫
        query_embedding = self.embedding_model.encode([query])
        results = self.vector_db.query(
            query_embeddings=query_embedding,
            where={"user_id": user_id},
            n_results=5
        )
        return results
```
