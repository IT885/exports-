class QwenKeywordProcessor:
    """使用Qwen系列模型進行關鍵字處理和內容整理"""
    
    def __init__(self, model_name=MODEL_NAME, device=DEVICE):
        print(f"初始化Qwen模型: {model_name}")
        self.tokenizer = AutoTokenizer.from_pretrained(model_name, trust_remote_code=True)

        """生成模型回應"""
        messages = [{"role": "user", "content": prompt}]
        input_text = self.tokenizer.apply_chat_template(
            messages, 
            tokenize=False, 
            add_generation_prompt=True
        )
        inputs = self.tokenizer(input_text, return_tensors="pt").to(self.model.device)
        
        with torch.no_grad():
            outputs = self.model.generate(
                **inputs, 
                max_new_tokens=max_length,
                do_sample=True,
                temperature=0.7,
                top_p=0.8
            )
        
        response = self.tokenizer.decode(outputs[0][inputs.input_ids.shape[1]:], skip_special_tokens=True)
        return response.strip()
    
    def expand_query(self, query):
        """使用模型擴展查詢關鍵字"""
        prompt = f"""請幫我提供5個與"{query}"相關的搜索關鍵詞組合，這些組合應該能幫助獲取更全面的信息。
只需列出關鍵詞，不要有任何解釋或編號。每個關鍵詞組合應該在一行。"""
        
        response = self.generate_response(prompt)

        return list(set(expanded_queries))[:5]  # 去重並限制數量
    
    def extract_key_sentences(self, text, query_keywords, top_n=5):
        """使用BM25算法提取與查詢最相關的句子"""
        # 簡單的句子切分
        sentences = re.split(r'[。！？!?.]', text)
        sentences = [s.strip() for s in sentences if len(s.strip()) > 10]
        
        if not sentences:
            return []
            
        # 使用BM25進行排序
        tokenized_sentences = [list(s) for s in sentences]  # 中文按字符切分
        bm25 = BM25Okapi(tokenized_sentences)
        
        # 將查詢關鍵詞轉換為列表
        if isinstance(query_keywords, str):
            query_keywords = query_keywords.split()
            
        # 計算每個句子的得分
        scores = bm25.get_scores(list("".join(query_keywords)))
        
        # 獲取得分最高的句子
        top_indices = np.argsort(scores)[::-1][:top_n]
        return [sentences[i] for i in top_indices if scores[i] > 0]
    
    def analyze_search_results(self, results, query):
        """分析搜索結果，提取關鍵信息和洞見"""
        results_text = ""

2. 對搜索結果的簡要總結（不超過3句話）

搜索結果：
{results_text}"""

        response = self.generate_response(prompt)
        return response

class MCPWebCrawler:
    """MCP服務器爬蟲工具，現整合Qwen模型處理"""
    
    def __init__(self, name="web_crawler", model_name=MODEL_NAME):
        self.name = name
        try:
            self.qwen_processor = QwenKeywordProcessor(model_name)
            print(f"已載入Qwen模型: {model_name}")
        except Exception as e:
            self.qwen_processor = None
            print(f"載入Qwen模型失敗: {e}")
            
    def search(self, query, engines=None, max_results=5, expand_queries=False):
        """搜索功能（MCP工具）"""
        if not engines:
            engines = ["google"]
            
        results = []
        for engine in engines:
            result = search_with_engine(
                query, 
                engine, 
                max_results,
                expand_queries=expand_queries,
                qwen_processor=self.qwen_processor
            )
            results.append(result)
            
        # 使用Qwen整合結果
        integrated_analysis = ""
        if self.qwen_processor:
            all_snippets = []
            for result in results:
                for item in result.get("results", []):
                    all_snippets.append(item.get("snippet", ""))
                    
            if all_snippets:
                prompt = f"""以下是關於"{query}"的多個搜索引擎結果的摘要，請整合這些內容並提供一個全面的總結，不超過100字：
                
{' '.join(all_snippets[:10])}"""
                integrated_analysis = self.qwen_processor.generate_response(prompt)
                
        return {
            "query": query,
            "timestamp": datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S'),
            "engines": engines,
            "results": results,
            "integrated_analysis": integrated_analysis
        }

        if self.qwen_processor:

            prompt = f"""請分析以下關鍵詞，並基於這些關鍵詞提供5-8個更多的相關搜索詞：
{', '.join(expanded_queries)}

只需列出關鍵詞，不要有任何解釋或編號。每個關鍵詞組合應該在一行。"""
            
            more_keywords = self.qwen_processor.generate_response(prompt).split('\n')
            all_keywords = list(set(expanded_queries + more_keywords))
        else:
            # 回退到傳統方法
            result = search_with_engine(query, "google", max_results=1)
            related_searches = result.get("related_searches", [])
            expanded_queries = self.expand_query_fallback(query)
            all_keywords = list(set(related_searches + expanded_queries))
            
        return {
            "query": query,
            "related_keywords": all_keywords
        }
        
    def expand_query_fallback(self, query):
        """傳統查詢擴展（在沒有Qwen模型時使用）"""
        expanded_queries = [query]  # 原始查詢
        
        # 基本分詞
        terms = query.split()
        if len(terms) > 1:
            expanded_queries.extend(terms)
            
        # 領域特定擴展
        domain_expansions = {
            "美食": ["餐廳", "食物", "料理", "小吃"],
            "旅遊": ["景點", "行程", "攻略", "住宿"],
            "科技": ["技術", "創新", "產品", "趨勢"],
            "資安": ["網路安全", "資訊安全", "駭客", "防護"],
        }
        
        # 根據領域生成新查詢
        for term in terms:
            if term in domain_expansions:
                for expansion in domain_expansions[term]:
                    expanded_queries.append(query.replace(term, expansion))
                    
        return list(set(expanded_queries))[:5]  # 去重並限制數量

def search_and_save_as_md(query, engines=None, max_results=5, debug=False, save_full_content=False, expand_queries=False, model_name=MODEL_NAME):
    """使用多個搜索引擎搜索並將結果保存為Markdown，整合Qwen模型分析"""
    print(f"開始搜索: {query}")
    
    # 創建results目錄（如果不存在）
    if not os.path.exists("markdown_results"):
        os.makedirs("markdown_results")
        
    timestamp = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
    filename = f"markdown_results/search_results_{clean_filename(query)}_{timestamp}.md"
    
    # 如果未指定搜索引擎，使用默認值
    if not engines:
        engines = ["google", "bing"]
        
    # 初始化Qwen處理器
    qwen_processor = None
    try:
        qwen_processor = QwenKeywordProcessor(model_name)
        print(f"已載入Qwen模型: {model_name}")
    except Exception as e:
        print(f"載入Qwen模型失敗，將使用傳統方法: {e}")
    
    # 使用線程池並行搜索
    all_results = []
    with concurrent.futures.ThreadPoolExecutor(max_workers=len(engines)) as executor:
        # 創建任務
        future_to_engine = {
            executor.submit(
                search_with_engine, 
                query, 
                engine, 
                max_results, 
                debug, 
                save_full_content, 
                expand_queries,
                qwen_processor
            ): engine
            for engine in engines
        }
        
        # 獲取結果
        for future in concurrent.futures.as_completed(future_to_engine):
            engine = future_to_engine[future]
            try:
                result = future.result()
                all_results.append(result)
                print(f"{engine} 搜索完成，找到 {result['results_count']} 個結果")
            except Exception as e:
                print(f"{engine} 搜索失敗: {e}")
    
    # 收集所有相關搜索和分析
    all_related_searches = []
    all_analyses = []
    for result in all_results:
        if "related_searches" in result:
            all_related_searches.extend(result["related_searches"])
        if "qwen_analysis" in result and result["qwen_analysis"]:
            all_analyses.append(result["qwen_analysis"])
            
    # 使用Qwen整合所有結果
    integrated_analysis = ""
    if qwen_processor and all_results:
        all_snippets = []
        for result in all_results:
            for item in result.get("results", []):
                all_snippets.append(item.get("snippet", ""))
                
        if all_snippets:
            prompt = f"""以下是關於"{query}"的多個搜索引擎結果的摘要，請整合這些內容並提供：
1. 5個關鍵發現或要點
2. 一個全面的總結（不超過150字）

{' '.join(all_snippets[:15])}"""
            integrated_analysis = qwen_processor.generate_response(prompt)
    
    # 保存為Markdown
    with open(filename, "w", encoding="utf-8") as f:
        f.write(f"# 多引擎搜索結果: {query}\n")
        f.write("=========================\n\n\n")
        
        f.write(f"搜索查詢: **{query}**\n\n")
        f.write(f"搜索時間: {datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S')}\n\n")
        f.write(f"搜索引擎: **{', '.join(engines)}**\n\n")
        
        # 統計總結果數
        total_results = sum(r["results_count"] for r in all_results)
        f.write(f"總結果數量: **{total_results}**\n\n")
        
        # 如果有Qwen整合分析，顯示在開頭
        if integrated_analysis:
            f.write("## Qwen整合分析\n\n")
            f.write(f"{integrated_analysis}\n\n")
            f.write("---\n\n")
        
        # 如果啟用了查詢擴展，顯示擴展的查詢
        if expand_queries:
            expanded_queries = []
            for result in all_results:
                if "expanded_queries" in result:
                    expanded_queries.extend(result["expanded_queries"])
            if expanded_queries:
                f.write("## 擴展查詢\n\n")
                for eq in set(expanded_queries):
                    f.write(f"- {eq}\n")
                f.write("\n")
        
        # 顯示相關搜索
        if all_related_searches:
            f.write("## 相關搜索\n\n")
            for rs in set(all_related_searches):
                f.write(f"- {rs}\n")
            f.write("\n")
            
        if total_results == 0:
            f.write("**未找到相關結果**\n\n")
            
        # 為每個搜索引擎寫入結果
        for engine_result in all_results:
            engine = engine_result["engine"]
            results = engine_result["results"]
            
            f.write(f"## {engine.upper()} 搜索結果 ({len(results)})\n\n")
            
            # 顯示該引擎的Qwen分析（如果有）
            if "qwen_analysis" in engine_result and engine_result["qwen_analysis"]:
                f.write("### Qwen分析\n\n")
                f.write(f"{engine_result['qwen_analysis']}\n\n")
                f.write("---\n\n")
            
            if "error" in engine_result and engine_result["error"]:
                f.write(f"**錯誤**: {engine_result['error']}\n\n")
                
            for i, result in enumerate(results):
                f.write(f"### {i+1}. {result['title']}\n\n")
                f.write(f"**鏈接**: [{result['link']}]({result['link']})\n\n")
                
                if "query" in result and result["query"] != query:
                    f.write(f"**使用查詢**: {result['query']}\n\n")
                    
                if result.get("author"):
                    f.write(f"**作者**: {result['author']}\n\n")
                    
                if result.get("publish_date"):
                    f.write(f"**發布日期**: {result['publish_date']}\n\n")
                    
                f.write(f"**摘要**: {result['snippet']}\n\n")
                
                if result.get("paragraphs") and len(result["paragraphs"]) > 0:
                    f.write("**相關內容**:\n\n")
                    for p in result["paragraphs"]:
                        f.write(f"> {p}\n\n")
                        
                f.write("---\n\n")
                
            f.write("\n")
            
        f.write("\n*搜索結果由Qwen增強型多引擎網絡爬蟲自動生成*")
    
    # 同時保存為JSON格式（方便程序讀取）
    json_filename = f"markdown_results/search_results_{clean_filename(query)}_{timestamp}.json"
    with open(json_filename, "w", encoding="utf-8") as f:
        json.dump({
            "query": query,
            "timestamp": datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S'),
            "engines": engines,
            "results": all_results,
            "related_searches": list(set(all_related_searches)),
            "integrated_analysis": integrated_analysis
        }, f, ensure_ascii=False, indent=2)
        
    print(f"搜索結果已保存到: {filename} 和 {json_filename}")
    return filename

## 關鍵詞處理流程

當啟用`expand_queries`參數時，程式會：

1. 透過`qwen_processor.expand_query()`函數將用戶輸入的查詢（如"牛座跟牡羊座在一起最喜歡做什麼"）轉換為更適合搜尋的關鍵詞組合[2][3]
2. 這個過程通過以下提示實現：
   ```
   請幫我提供5個與"{query}"相關的搜索關鍵詞組合，這些組合應該能幫助獲取更全面的信息
   ```
3. 系統會保留原始查詢，同時添加AI生成的擴展關鍵詞，最多保留5個獨特的查詢[4]

## 搜尋結果整理

搜尋完成後，程式會：

1. 使用`qwen_processor.analyze_search_results()`函數分析搜尋結果[3]
2. 提取3-5個主要關鍵詞或術語
3. 生成簡要總結（不超過3句話）
4. 如果使用多個搜尋引擎，還會整合所有結果並提供5個關鍵發現和全面總結[4]

這種設計能有效解決直接使用原始語句搜索可能產生的問題，讓結果更加精確和相關。
